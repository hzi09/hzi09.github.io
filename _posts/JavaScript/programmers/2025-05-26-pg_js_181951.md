const readline = require("readline");

  input: process.stdin,
  output: process.stdout,
});

let input = [];

rl.on("line", function (line) {
  input = line.split(" ").map(Number);
  rl.close();
}).on("close", function () {
  const a = input[0];
  const b = input[1];

  console.log(`a = ${a}`);
  console.log(`b = ${b}`);
});

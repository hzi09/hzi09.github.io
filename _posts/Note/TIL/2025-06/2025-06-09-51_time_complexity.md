def compress_string(s):
    result = []
    count = 1

    for i in range(1, len(s)):    # O(n)
        if s[i] == s[i-1]:
            count += 1
        else:
            result.append(s[i-1] + str(count))  # O(1)
            count = 1
    result.append(s[-1] + str(count))          # O(1)

    compressed = ''.join(result)               # O(n)
    return compressed if len(compressed) < len(s) else s
